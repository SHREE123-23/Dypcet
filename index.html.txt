   ANIMATED COUNTER
══════════════════════════════════════ */
function animateCounter(el, target, suffix='', duration=1200){
  let start=0, step=target/60;
  const timer=setInterval(()=>{
    start=Math.min(start+step, target);
    el.textContent=Math.round(start)+(suffix||'');
    if(start>=target) clearInterval(timer);
  },duration/60);
}

/* ══════════════════════════════════════
   SMART BADGES
══════════════════════════════════════ */
function getSmartBadges(item){
  const badges=[];
  const allRatings=listings.map(l=>avgRating(l.reviews));
  const maxRating=Math.max(...allRatings);
  const minPrice=Math.min(...listings.map(l=>l.price));
  const minDist=Math.min(...listings.map(l=>l.dist));
  const reviewCounts=listings.map(l=>l.reviews.length);
  const maxReviews=Math.max(...reviewCounts);

  if(item.reviews.length===maxReviews && maxReviews>0) badges.push('<span class="smart-badge sb-popular">🔥 Most Popular</span>');
  if(item.price===minPrice) badges.push('<span class="smart-badge sb-value">💰 Best Value</span>');
  if(item.dist===minDist) badges.push('<span class="smart-badge sb-nearest">📍 Nearest</span>');
  if(avgRating(item.reviews)===maxRating && item.reviews.length>0) badges.push('<span class="smart-badge sb-rated">⭐ Highest Rated</span>');
  return badges.join('');
}

/* ══════════════════════════════════════
   HELPERS
══════════════════════════════════════ */
function starsHTML(avg,count){
  if(!count) return '<span style="font-size:0.72rem;color:var(--muted);">No reviews yet</span>';
  let s='';for(let i=1;i<=5;i++) s+=`<span style="color:${i<=Math.round(avg)?'var(--gold)':'#d1d5db'}">★</span>`;
  return `<span style="font-size:0.85rem;">${s}</span> <span style="font-size:0.72rem;color:var(--muted);font-weight:700;">${avg.toFixed(1)} (${count})</span>`;
}
function avgRating(reviews){
  if(!reviews.length) return 0;
  return reviews.reduce((a,r)=>a+r.rating,0)/reviews.length;
}

/* ══════════════════════════════════════
   FILTER & SORT
══════════════════════════════════════ */
function doFilter(){
  const s=document.getElementById('f-search').value.toLowerCase();
  const t=document.getElementById('f-type').value;
  const d=document.getElementById('f-dist').value;
  const b=document.getElementById('f-budget').value;
  const g=document.getElementById('f-gender').value;
  const beds=document.getElementById('f-beds').value;
  const av=document.getElementById('f-avail').value;
  const sort=document.getElementById('f-sort').value;

  let filtered=listings.filter(i=>
    i.name.toLowerCase().includes(s) &&
    (t==='all'||i.type===t) &&
    (d==='all'||i.dist<=parseInt(d)) &&
    (b==='all'||i.price<=parseInt(b)) &&
    (g==='all'||i.gender===g) &&
    (beds==='all'||i.beds.includes(parseInt(beds))||i.type==='Mess') &&
    (av==='all'||(av==='open'&&i.available)||(av==='full'&&!i.available))
  );

  // Sort
  if(sort==='price-asc') filtered.sort((a,b)=>a.price-b.price);
  else if(sort==='price-desc') filtered.sort((a,b)=>b.price-a.price);
  else if(sort==='dist-asc') filtered.sort((a,b)=>a.dist-b.dist);
  else if(sort==='rating-desc') filtered.sort((a,b)=>avgRating(b.reviews)-avgRating(a.reviews));

  document.getElementById('result-num').textContent=filtered.length;
  if(currentView==='grid') renderCards(filtered);
  renderMapPins(filtered);
}

function quizSelect(step, key, val){
  quizAnswers[key]=val;
  document.querySelectorAll(`#qs-${step} .quiz-opt`).forEach(o=>o.classList.remove('selected'));
  event.currentTarget.classList.add('selected');
  setTimeout(()=>{
    if(step<totalQuizSteps){quizStep=step+1;renderQuizStep();}
    else{quizStep=totalQuizSteps+1;renderQuizStep();computeQuizResults();}
  },380);
}
function quizBack(){if(quizStep>1){quizStep--;renderQuizStep();}}
function computeQuizResults(){
  const budget=parseInt(quizAnswers.budget)||9999;
  const dist=parseInt(quizAnswers.dist)||9999;
  const type=quizAnswers.type||'all';
  const priority=quizAnswers.priority||'';
  let scored=listings.map(item=>{
    let score=0;
    if(item.price<=budget) score+=30;
    if(item.dist<=dist) score+=25;
    if(type==='all'||item.type===type) score+=20;
    if(priority==='wifi'&&item.amenities.includes('wifi')) score+=15;
    if(priority==='mess'&&item.amenities.includes('mess')) score+=15;
    if(priority==='security'&&(item.amenities.includes('security')||item.amenities.includes('cctv'))) score+=15;
    if(priority==='rating') score+=avgRating(item.reviews)*3;
    if(item.available) score+=5;
    return {...item, score};
  }).filter(i=>i.score>0).sort((a,b)=>b.score-a.score).slice(0,3);
  const labels=['🥇 Best Match','🥈 Great Option','🥉 Also Consider'];
  document.getElementById('quiz-results-list').innerHTML=scored.map((item,i)=>`
    <div class="quiz-result-card" onclick="closeQuiz();openDetail(${item.id})">
      <div class="rank">${labels[i]||'Option'}</div>
      <div style="font-weight:800;font-size:0.95rem;color:var(--text);font-family:'Syne',sans-serif;">${item.name}</div>
      <div style="font-size:0.78rem;color:var(--muted);margin-top:4px;">₹${item.price.toLocaleString()}/mo · ${item.dist}m · ${item.type} · ${item.gender}</div>
      <div style="margin-top:8px;">${starsHTML(avgRating(item.reviews),item.reviews.length)}</div>
      <div style="font-size:0.72rem;color:var(--muted);margin-top:6px;font-weight:700;">Match score: ${item.score}/100</div>
    </div>`).join('');
}
function applyQuizFilters(){

  if(quizAnswers.budget) document.getElementById('f-budget').value=quizAnswers.budget;
  if(quizAnswers.dist&&quizAnswers.dist!=='9999') document.getElementById('f-dist').value=quizAnswers.dist;
  doFilter();

}
